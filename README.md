# docker
docker image
# Build an image from a Dockerfile:
> docker build -t image_name path_to_dockerfile  
  >> Example:
> docker build -t myapp .
# List all local images:
> docker images
 >> Example:
> docker image ls
# pull an image from docker hub:
> docker pull image_name:tag
 >> Example:
> docker pull nginx:latest
